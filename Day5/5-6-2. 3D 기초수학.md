float speed;

void Start {
  speed = 3.0f;
}

void Update() {
  float h = Input.GetAxis("Horizontal"); // AD. -1 ~ 1. x 방향
  float v = Input.GetAxis("Vertical"); // WS. -1 ~ 1. z 방향
  Vector3 move = new Vector3(h, 0, v); // x, z 방향 움직임
  controller.Move(move * speed * Time.deltaTime);
}
